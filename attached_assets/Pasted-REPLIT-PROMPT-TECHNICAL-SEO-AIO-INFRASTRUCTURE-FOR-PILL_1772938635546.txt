PART 3 — JSON-LD STRUCTURED DATA
Create /components/JsonLd.tsx:

export default function JsonLd({ data }: { data: object }) {
  return (
    <script
      type="application/ld+json"
      dangerouslySetInnerHTML={{ __html: JSON.stringify(data) }}
    />
  )
}


This must render server-side in static HTML — not via useEffect or client-side JavaScript.

PART 5 — SITEMAP
Create next-sitemap.config.js in root:

/** @type {import('next-sitemap').IConfig} */
module.exports = {
  siteUrl: 'https://pillardrugclub.com',
  generateRobotsTxt: false,
  changefreq: 'weekly',
  priority: 0.7,
  sitemapSize: 5000,
  exclude: ['/api/*', '/dashboard/*', '/account/*'],
  additionalPaths: async (config) => [
    await config.transform(config, '/'),
    await config.transform(config, '/faq'),
    await config.transform(config, '/pharmacy-membership-vs-goodrx'),
