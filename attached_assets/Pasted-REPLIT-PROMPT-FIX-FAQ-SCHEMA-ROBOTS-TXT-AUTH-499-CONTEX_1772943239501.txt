Fix the loadUser function to skip the API call entirely if no cached user exists:

const loadUser = async () => {
  try {
    console.log('[AuthContext] Starting loadUser...')
    
    const cachedUser = localStorage.getItem('user')
    console.log('[AuthContext] localStorage user:', cachedUser ? 'found' : 'not found')
    
    // Skip API call if no cached user — prevents timeout on unauthenticated requests
    if (!cachedUser) {
      setIsLoading(false)
      return
    }
    
    const response = await fetch('/api/auth/user')
    console.log('[AuthContext] Server response status:', response.status)
    
    if (!response.ok) {
      console.warn('[AuthContext] Server error while checking auth, keeping cached user:', response.status)
      setIsLoading(false)
      return
    }
    
    const data = await response.json()
    setUser(data)
  } catch (err) {
    console.warn('[AuthContext] Auth check failed:', err)
  } finally {
    setIsLoading(false)
  }
}
