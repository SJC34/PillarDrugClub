Square uses a two-step flow:
1. Frontend: Square Web Payments SDK tokenizes the card (never touches your server)
2. Backend: You charge the token via Square Payments API

3a. server/services/payment-service.ts
Create or overwrite with a Square payment service:

/**
 * Payment Service — Square
 * Docs: https://developer.squareup.com/docs/payments-api/take-payments
 *
 * Required env vars:
 *   SQUARE_ACCESS_TOKEN   — from Square Developer Dashboard
 *   SQUARE_LOCATION_ID    — from Square Developer Dashboard
 *   SQUARE_ENVIRONMENT    — "sandbox" | "production"
 */
import { Client, Environment } from 'square';

function getClient() {
  const token = process.env.SQUARE_ACCESS_TOKEN;
  const env   = process.env.SQUARE_ENVIRONMENT === 'production'
    ? Environment.Production
    : Environment.Sandbox;
  if (!token) throw new Error('SQUARE_ACCESS_TOKEN not configured.');
  return new Client({ accessToken: token, environment: env });
}

export function getLocationId(): string {
  const id = process.env.SQUARE_LOCATION_ID;
  if (!id) throw new Error('SQUARE_LOCATION_ID not configured.');
  return id;
}

export interface ChargeInput {
  sourceId:      string;   // Tokenized card nonce from Square Web Payments SDK
  amountCents:   number;
  currency:      string;   // 'USD'
  customerId:    string;   // Your internal user ID
  customerEmail: string;
  note:          string;
}

export interface ChargeResult {
  success:       boolean;
  paymentId?:    string;
  receiptUrl?:   string;
  error?:        string;
}

export async function chargePayment(input: ChargeInput): Promise<ChargeResult> {
  try {
    const { paymentsApi } = getClient();
    const { result } = await paymentsApi.createPayment({
      sourceId:         input.sourceId,
      idempotencyKey:   `${input.customerId}-${Date.now()}`,
      amountMoney:      { amount: BigInt(input.amountCents), currency: input.currency },
      locationId:       getLocationId(),
      note:             input.note,
      buyerEmailAddress: input.customerEmail,
    });
    return {
      success:    true,
      paymentId:  result.payment?.id,
      receiptUrl: result.payment?.receiptUrl ?? undefined,
    };
  } catch (err: any) {
    return { success: false, error: err?.errors?.[0]?.detail || String(err) };
  }
}

export async function refundPayment(paymentId: string, amountCents: number, reason: string): Promise<ChargeResult> {
  try {
    const { refundsApi } = getClient();
    const { result } = await refundsApi.refundPayment({
      paymentId,
      idempotencyKey: `refund-${paymentId}-${Date.now()}`,
      amountMoney:    { amount: BigInt(amountCents), currency: 'USD' },
      reason,
    });
    return { success: true, paymentId: result.refund?.id };
  } catch (err: any) {
    return { success: false, error: err?.errors?.[0]?.detail || String(err) };
  }
}

export function isPaymentConfigured(): boolean {
  return !!(process.env.SQUARE_ACCESS_TOKEN && process.env.SQUARE_LOCATION_ID);
}

3c. client/src/pages/RegisterPage.tsx
Remove all @stripe/react-stripe-js and @stripe/stripe-js imports.
Load Square Web Payments SDK dynamically and replace the card form with:

function SquarePaymentForm({ onTokenize }: { onTokenize: (token: string) => void }) {
  const cardContainerRef = useRef<HTMLDivElement>(null);
  const [card, setCard] = useState<any>(null);

  useEffect(() => {
    const appId      = import.meta.env.VITE_SQUARE_APP_ID;
    const locationId = import.meta.env.VITE_SQUARE_LOCATION_ID;
    if (!appId || !locationId || !cardContainerRef.current) return;
    (async () => {
      const payments     = (window as any).Square.payments(appId, locationId);
      const cardInstance = await payments.card();
      await cardInstance.attach(cardContainerRef.current!);
      setCard(cardInstance);
    })();
  }, []);

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault();
    if (!card) return;
    const result = await card.tokenize();
    if (result.status === 'OK') onTokenize(result.token);
  }

  return (
    <form onSubmit={handleSubmit}>
      <div ref={cardContainerRef} id="card-container" />
      <button type="submit">Pay Now</button>
    </form>
  );
}
